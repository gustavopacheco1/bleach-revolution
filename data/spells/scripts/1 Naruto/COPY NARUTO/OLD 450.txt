local combat = createCombatObject()
setCombatParam(combat, COMBAT_PARAM_HITCOLOR, COLOR_RED)
setCombatParam(combat, COMBAT_PARAM_TYPE, COMBAT_PHYSICALDAMAGE)
setCombatParam(combat, COMBAT_PARAM_DISTANCEEFFECT, 108)
setCombatFormula(combat, COMBAT_FORMULA_LEVELMAGIC, -50.3, 1, -50.3, 1)


 
function onCastSpell(cid, var) 
	local exausted = 1.0 
	local storage = 250000
	local effect = 82
	local target = getCreatureTarget(cid)
	
	if exhaustion.check(cid, storage) then
		doPlayerSendCancel(cid, "You are exhausted")
	return false
  end

	for i = 1, 5 do
		if isCreature(target) then
			addEvent(function()
			local PosTarget = {x = getCreaturePosition(target).x, y = getCreaturePosition(target).y, z = getCreaturePosition(target).z}    
			local position1 = {x=getThingPosition(getCreatureTarget(cid)).x, y=getThingPosition(getCreatureTarget(cid)).y, z=getThingPosition(getCreatureTarget(cid)).z}	
			doSendMagicEffect(PosTarget, effect)
			doSendMagicEffect(position1, 121)
			return doCombat(cid, combat, var)
        end, 500*i)
    end
end
exhaustion.set(cid, storage, exausted)
return TRUE
end
